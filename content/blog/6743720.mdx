问：

// 协议基 类  
struct p\_base  
{  
p\_base(int protocol\_code,int protocol\_len):p\_name(protocol\_code),p\_len(protocol\_len)  
{}  
int p\_name;  
int p\_len;  
};

答：

这个就相当于class p\_base  
{  
public:  
int p\_name;  
int p\_len;  
public:  
p\_base(int protocol\_code,int protocol\_len)  
{  
p\_name = protocol\_code;  
p\_len = protocol\_len;  
}  
};
