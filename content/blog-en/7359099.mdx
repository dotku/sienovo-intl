The requirement is to reverse a singly linked list with a head node `List head`.

Analysis:

  1). If the linked list is empty or contains only one element, return directly;

  2). Set up two adjacent pointers, `p` and `q`. The node pointed to by `p` will become the successor of the node pointed to by `q`;

  3). Repeat step 2) until `q` is null.

  4). Adjust the head and tail of the linked list.

1.  #include <stdio.h>  
2.  #include <stdlib.h>  

4.  typedef struct tagListNode{  
5.      int data;  
6.      struct tagListNode\* next;  
7.  }ListNode, \*List;  

9.  void PrintList(List head);  
10.  List ReverseList(List head);  

12.  int main()  
13.  {  
14.      //Allocate head node for the linked list  
15.      ListNode \*head;  
16.      head = (ListNode\*)malloc(sizeof(ListNode));  
17.      head->next = NULL;  
18.      head->data = -1;  

20.      //Add [1,10] to the linked list  
21.      int i;  
22.      ListNode \*p, \*q;  
23.      p = head;  
24.      for(int i = 1; i <= 10; i++)  
25.      {  
26.          q = (ListNode \*)malloc(sizeof(ListNode));  
27.          q->data = i;  
28.          q->next = NULL;  
29.          p->next = q;  
30.          p = q;          
31.      }  

33.      PrintList(head);           /* Output original linked list */  
34.      head = ReverseList(head);  /* Reverse linked list */  
35.      PrintList(head);           /* Output reversed linked list */  
36.      return 0;  
37.  }  

39.  List ReverseList(List head)  
40.  {  
41.      if(head->next == NULL || head->next->next == NULL)    
42.      {  
43.         return head;   /* If the linked list is empty or has only one element, return directly */  
44.      }  

46.      ListNode \*t = NULL,  
47.               \*p = head->next,  
48.               \*q = head->next->next;  
49.      while(q != NULL)  
50.      {          
51.        t = q->next;  
52.        q->next = p;  
53.        p = q;  
54.        q = t;  
55.      }  

57.      /* At this point, q points to the last element of the original list, which is also the head element of the reversed list */  
58.      head->next->next = NULL;  /* Set the linked list tail */  
59.      head->next = p;           /* Adjust the linked list head */  
60.      return head;  
61.  }  

63.  void PrintList(List head)  
64.  {  
65.      ListNode\* p = head->next;  
66.      while(p != NULL)  
67.      {  
68.          printf("%d ", p->data);  
69.          p = p->next;  
70.      }  
71.      printf("/n");  
72.  }
