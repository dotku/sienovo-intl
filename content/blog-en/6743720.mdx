Question:

// Protocol base class  
struct p_base  
{  
    p_base(int protocol_code, int protocol_len) : p_name(protocol_code), p_len(protocol_len)  
    {}  
    int p_name;  
    int p_len;  
};

Answer:

This is equivalent to:

class p_base  
{  
public:  
    int p_name;  
    int p_len;  

public:  
    p_base(int protocol_code, int protocol_len)  
    {  
        p_name = protocol_code;  
        p_len = protocol_len;  
    }  
};
